Setting volume (set volume in "volume" variable) 

 #include <sys/ioctl.h>
 #include <sys/soundcard.h>
 #include <stdio.h>

 [...]
 long mixer;
 int realVol = (this->volume << 8) + this->volume; /* 0 < volume < 255 */
 mixer = open("/dev/mixer", O_RDWR);

 if (mixer != -1) {
    ioctl(mixer, SOUND_MIXER_WRITE_VOLUME , &realVol);
    close(mixer);
 }
